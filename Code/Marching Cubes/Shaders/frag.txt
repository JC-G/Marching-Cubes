//basic test fragment shader
#version 330

precision highp float;
layout(location = 0) out vec4 color;

in vec4 vertexPosition_worldSpace;

void main() {
	color.a = 1;
	//color corners of test cube white, rest black
	if(min(abs(vertexPosition_worldSpace.x),min(abs(vertexPosition_worldSpace.y),abs(vertexPosition_worldSpace.z))) > 0.9)
        color.rgb = vec3(1.0);
    else
        color.rgb = vec3(0.0);
}
